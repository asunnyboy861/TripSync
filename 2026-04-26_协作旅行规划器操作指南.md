### 2.2 TREK项目深度分析（核心参考项目）

**技术栈**:
- 后端: Node.js 22 + Express + SQLite + WebSocket（实时同步）
- 前端: React 18 + Vite + TypeScript + Tailwind CSS + Leaflet/Mapbox GL
- 状态管理: Zustand
- 认证: JWT + OAuth 2.1 + OIDC + TOTP MFA
- 天气: Open-Meteo（无需API Key）
- 地图: Leaflet + Mapbox GL（3D建筑、地形、照片标记、聚类、路线可视化）

**核心功能清单**（可作为iOS版本功能设计蓝图）:
1. 拖拽式行程规划器 — 按天组织地点，支持跨天移动
2. 交互式地图 — 3D地图、路线可视化、地点聚类
3. 地点搜索 — Google Places / OpenStreetMap
4. 实时同步 — WebSocket即时同步所有用户
5. 多用户协作 — 邀请链接、角色权限
6. 预算追踪 — 分类支出、饼图、多人/每日分摊、多币种
7. 打包清单 — 分类、模板、成员分配
8. 文档管理 — 附件/票据/PDF
9. 协作套件 — 群聊、共享笔记、投票、每日签到
10. 旅行日记 — 杂志风格、照片、地图、心情
11. AI集成 — MCP服务器，150+工具

**二次开发策略**: TREK的AGPL v3协议要求修改后开源。因此我们的策略是：
- **参考架构设计**而非直接使用代码
- **复用核心概念**（实时协作架构、数据模型、功能设计）
- **iOS原生重构**：用Swift + SwiftUI + CloudKit重新实现核心功能
- **差异化**：加入无登录协作（TREK需要注册）、投票决策、滑动选择等独特功能

#### 3.2.2 无登录协作核心（ShareLink机制）

```swift
import Foundation
import CloudKit

/// 无登录分享管理器
actor ShareLinkManager {
    
    /// 生成旅行分享链接
    /// - Parameters:
    ///   - trip: 旅行对象
    ///   - accessLevel: 访问级别
    ///   - expiry: 链接过期时间（可选）
    /// - Returns: 分享URL
    func generateShareLink(
        for trip: Trip,
        accessLevel: AnonymousVisitor.AccessLevel = .editor,
        expiry: Date? = nil
    ) async throws -> URL {
        
        // 1. 生成唯一分享令牌
        let token = UUID().uuidString.replacingOccurrences(of: "-", with: "")
            .prefix(12).lowercased()
        
        // 2. 创建CloudKit分享记录
        let shareRecord = CKRecord(recordType: "TripShare")
        shareRecord["tripID"] = trip.id.uuidString
        shareRecord["shareToken"] = token
        shareRecord["accessLevel"] = accessLevel.rawValue
        shareRecord["isActive"] = true
        
        if let expiry = expiry {
            shareRecord["expiryDate"] = expiry
        }
        
        // 3. 保存到CloudKit Public Database
        let container = CKContainer(identifier: "iCloud.com.tripsync.app")
        let publicDB = container.publicCloudDatabase
        try await publicDB.save(shareRecord)
        
        // 4. 构建Universal Link
        var components = URLComponents(string: "https://tripsync.app/join")
        components?.queryItems = [
            URLQueryItem(name: "t", value: String(token)),
            URLQueryItem(name: "a", value: accessLevel.rawValue)
        ]
        
        guard let url = components?.url else {
            throw ShareError.invalidURL
        }
        
        return url
    }
    
    /// 通过分享令牌访问旅行（无需登录）
    func accessTripViaToken(_ token: String) async throws -> (Trip, AnonymousVisitor) {
        let container = CKContainer(identifier: "iCloud.com.tripsync.app")
        let publicDB = container.publicCloudDatabase
        
        // 1. 查找分享记录
        let predicate = NSPredicate(format: "shareToken == %@ AND isActive == 1", token)
        let query = CKQuery(recordType: "TripShare", predicate: predicate)
        
        let results = try await publicDB.records(matching: query)
        guard let shareRecord = results.matchResults.first?.1,
              case .success(let record) = shareRecord else {
            throw ShareError.tokenNotFound
        }
        
        // 2. 检查过期
        if let expiry = record["expiryDate"] as? Date, expiry < Date() {
            throw ShareError.tokenExpired
        }
        
        // 3. 创建匿名访客身份
        let visitor = AnonymousVisitor(
            displayName: generateRandomName(),
            avatarEmoji: randomEmoji(),
            shareToken: token,
            accessLevel: AnonymousVisitor.AccessLevel(rawValue: record["accessLevel"] as? String ?? "viewer") ?? .viewer,
            lastActiveAt: Date()
        )
        
        // 4. 获取旅行数据
        guard let tripID = record["tripID"] as? String,
              let uuid = UUID(uuidString: tripID) else {
            throw ShareError.invalidData
        }
        
        let tripRecord = try await publicDB.record(for: CKRecord.ID(recordName: uuid.uuidString))
        let trip = try parseTrip(from: tripRecord)
        
        return (trip, visitor)
    }
    
    private func generateRandomName() -> String {
        let adjectives = ["Happy", "Sunny", "Brave", "Chill", "Witty", "Swift", "Cosmic", "Dreamy"]
        let nouns = ["Traveler", "Explorer", "Wanderer", "Voyager", "Nomad", "Adventurer", "Pilot", "Scout"]
        let number = Int.random(in: 1...99)
        return "\(adjectives.randomElement()!) \(nouns.randomElement()!) \(number)"
    }
    
    private func randomEmoji() -> String {
        let emojis = ["🌍", "✈️", "🏖️", "🗺️", "🎒", "🏕️", "🚀", "🌊", "🏔️", "🌴"]
        return emojis.randomElement()!
    }
}

enum ShareError: LocalizedError {
    case invalidURL
    case tokenNotFound
    case tokenExpired
    case invalidData
    
    var errorDescription: String? {
        switch self {
        case .invalidURL: return "Invalid share link"
        case .tokenNotFound: return "This share link doesn't exist"
        case .tokenExpired: return "This share link has expired"
        case .invalidData: return "Could not load trip data"
        }
    }
}
```

#### 3.2.3 实时同步管理器（CloudKit + 本地缓存）

```swift
import CloudKit
import SwiftUI
import Combine

/// 实时同步管理器
@Observable
final class SyncManager {
    var isSyncing = false
    var lastSyncTime: Date?
    var connectionStatus: ConnectionStatus = .disconnected
    
    enum ConnectionStatus {
        case connected
        case syncing
        case disconnected
        case offline
    }
    
    private let container = CKContainer(identifier: "iCloud.com.tripsync.app")
    private var subscriptions: [CKSubscription] = []
    private var cancellables = Set<AnyCancellable>()
    
    /// 订阅旅行变更
    func subscribeToTripChanges(tripID: UUID) async throws {
        let predicate = NSPredicate(format: "tripID == %@", tripID.uuidString)
        
        // 活动变更订阅
        let activitySubscription = CKQuerySubscription(
            recordType: "Activity",
            predicate: predicate,
            subscriptionID: CKSubscription.ID("trip-activities-\(tripID.uuidString)"),
            options: [.firesOnRecordCreation, .firesOnRecordUpdate, .firesOnRecordDeletion]
        )
        
        let notificationInfo = CKSubscription.NotificationInfo()
        notificationInfo.shouldSendContentAvailable = true
        activitySubscription.notificationInfo = notificationInfo
        
        // 投票变更订阅
        let pollSubscription = CKQuerySubscription(
            recordType: "Poll",
            predicate: predicate,
            subscriptionID: CKSubscription.ID("trip-polls-\(tripID.uuidString)"),
            options: [.firesOnRecordCreation, .firesOnRecordUpdate]
        )
        pollSubscription.notificationInfo = notificationInfo
        
        let publicDB = container.publicCloudDatabase
        try await publicDB.save(activitySubscription)
        try await publicDB.save(pollSubscription)
    }
    
    /// 离线优先写入：本地先保存，网络恢复后同步
    func saveActivity(_ activity: Activity, for tripID: UUID) async throws {
        // 1. 立即保存到本地SwiftData
        let localActivity = activity
        
        // 2. 尝试同步到CloudKit
        connectionStatus = .syncing
        isSyncing = true
        
        do {
            let record = try createCKRecord(from: activity, tripID: tripID)
            let publicDB = container.publicCloudDatabase
            try await publicDB.save(record)
            
            connectionStatus = .connected
            lastSyncTime = Date()
        } catch {
            // 网络失败：标记为待同步
            connectionStatus = .offline
            // 将保存到本地队列，等网络恢复后重试
            try await saveToPendingQueue(activity: activity, tripID: tripID)
        }
        
        isSyncing = false
    }
    
    private func createCKRecord(from activity: Activity, tripID: UUID) throws -> CKRecord {
        let record = CKRecord(recordType: "Activity")
        record["tripID"] = tripID.uuidString
        record["title"] = activity.title
        record["latitude"] = activity.latitude
        record["longitude"] = activity.longitude
        record["category"] = activity.category.rawValue
        record["order"] = activity.order
        record["addedBy"] = activity.addedBy
        if let startTime = activity.startTime { record["startTime"] = startTime }
        if let address = activity.address { record["address"] = address }
        return record
    }
    
    private func saveToPendingQueue(activity: Activity, tripID: UUID) async throws {
        // 保存到本地待同步队列

            recordType: "Activity",
            tripID: tripID,
```

## 十、技术风险与缓解

| 风险 | 影响 | 缓解措施 |
|------|------|---------|
| CloudKit免费额度不足 | 同步延迟/失败 | 监控API调用量，优化数据同步频率，增量同步 |
| 无登录滥用 | 恶意编辑/删除 | ShareToken有过期机制，管理员可锁定内容，操作审计日志 |
| 离线冲突 | 数据不一致 | Last-Write-Wins + 冲突提示UI |
| MapKit搜索限制 | 地点搜索不完整 | 备用: MKLocalSearch → Apple Maps Server API（付费） |
| App Store审核 | 无登录功能被质疑 | 准备详细说明文档，强调安全措施 |
